        // DATE HELPERS
        formatDate(d) {
            const date = new Date(d);
            const dd = String(date.getDate()).padStart(2, '0');
            const mm = String(date.getMonth() + 1).padStart(2, '0');
            const yyyy = date.getFullYear();
            return `${dd}/${mm}/${yyyy}`;
        },

        formatTime(d) {
            return new Date(d).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
        },

        formatDateTime(d) {
            return this.formatDate(d) + ' ' + this.formatTime(d);
        },

        getWeekStart(d) {
            const date = new Date(d);
            const day = date.getDay();
            date.setDate(date.getDate() - day);
            date.setHours(0, 0, 0, 0);
            return date;
        },

        async clockAction(type) {
            const emp = this.data.employees.find(e => e.id === this.data.selectedEmployeeId);
            if (!emp) return;
            // Prevent inactive employees from clocking in/out
            if (emp.active === false) {
                this.uiAlert('This profile is inactive. Please contact your manager.');
                this.navigate('employee-list');
                return;
            }

            if (this.data.isClockActionRunning) return;
            this.data.isClockActionRunning = true;
            this.showLoader('Saving clock action...');

            const btn = document.getElementById('action-btn');

            // If already in this state, ignore (prevents accidental double clicks)
            if ((type === 'in' && emp.status === 'in') || (type === 'out' && emp.status === 'out')) {
                this.data.isClockActionRunning = false;
                return;
            }

            // Verification based on settings (camera only here)
            let verificationPhoto = null;

            if (this.data.cameraVerificationEnabled) {
                // Camera capture mode
                verificationPhoto = await this.captureVerificationPhoto();
                if (!verificationPhoto) {
                    // User cancelled or camera failed
                    this.hideLoader();
                    this.data.isClockActionRunning = false;
                    return;
                }
            }
            
            // Check location restriction (GPS/IP) only if any restriction is enabled
            if (this.data.gpsRestrictionEnabled || this.data.ipRestrictionEnabled) {
                btn.innerHTML = '<i data-lucide="loader" class="w-5 h-5 inline mr-2 animate-spin"></i>Verifying Location...';
                btn.disabled = true;
                btn.className = 'w-full py-4 rounded-xl font-bold text-lg bg-slate-600 cursor-wait';
                lucide.createIcons();

                const locationCheck = await this.checkLocationRestriction();
                
                if (!locationCheck.allowed) {
                    btn.innerHTML = type === 'in' ? '<i data-lucide="log-in" class="w-5 h-5 inline mr-2"></i>Clock In' : '<i data-lucide="log-out" class="w-5 h-5 inline mr-2"></i>Clock Out';
                    btn.disabled = false;
                    btn.className = type === 'in' 
                        ? 'w-full py-4 rounded-xl font-bold text-lg transition-all active:scale-95 bg-green-500 hover:bg-green-600'
                        : 'w-full py-4 rounded-xl font-bold text-lg transition-all active:scale-95 bg-red-500 hover:bg-red-600';
                    btn.onclick = () => this.clockAction(type);
                    lucide.createIcons();
                    
                    this.uiAlert(`🙅 Nope, can't do that!\n\n${locationCheck.reason}\n\n🏪 Come to the store first, then try again! We're watching... 👀`);
                    this.hideLoader();
                    this.data.isClockActionRunning = false;
                    return;
                }
            }

            const now = new Date().toISOString();

            // Record ONLY the event times (multiple in/out per day supported)
            const historyRecord = {
                employeeId: emp.id,
                employeeName: emp.name,
                employeeRole: emp.role,
                action: type === 'in' ? 'Clocked In' : 'Clocked Out',
                timestamp: now
            };
            
            // Attach verification photo if captured (camera mode)
            if (verificationPhoto) {
                historyRecord.verificationPhoto = verificationPhoto;
            }

            if (type === 'in') {
                emp.status = 'in';
                emp.lastAction = now;
            } else {
                emp.status = 'out';
                emp.lastAction = now;
            }
            
            this.data.history.unshift(historyRecord);

            this.saveData();
            this.appendHistoryToSupabase(historyRecord);
            this.uiToast(type === 'in' ? 'Clocked In' : 'Clocked Out', 'success');

            btn.innerHTML = type === 'in' ? '✓ Clocked In Successfully!' : '✓ Clocked Out Successfully!';
            btn.disabled = true;
            btn.className = 'w-full py-4 rounded-xl font-bold text-lg bg-slate-700 cursor-not-allowed';
            setTimeout(() => this.navigate('employee-list'), 1500);
            this.hideLoader();
            this.data.isClockActionRunning = false;
        },

        // Prompt employee for 4-digit PIN and verify against stored pinHash
        promptEmployeePin(emp) {
            return new Promise((resolve) => {
                const container = document.getElementById('modal-container');
                const content = document.getElementById('modal-content');
                if (!container || !content) return resolve(false);

                const title = `Enter PIN for ${emp.name}`;
                content.innerHTML = `
                    <div class="glass-card rounded-2xl p-6 w-full max-w-sm">
                        <h3 class="text-lg font-bold mb-2">${title}</h3>
                        <p class="text-white/60 text-sm mb-4">Please enter your 4-digit password to continue.</p>
                        <div class="mb-4">
                            <label class="block text-white/60 text-xs mb-1">Password (4-digit PIN)</label>
                            <div class="relative">
                                <input id="pin-verify" type="password" inputmode="numeric" maxlength="4" pattern="\\d{4}" class="w-full input-dark rounded-lg px-3 py-2 pr-10 text-sm" placeholder="••••" />
                                <button type="button" id="pin-verify-toggle" class="absolute right-2 top-1/2 -translate-y-1/2 p-1 rounded hover:bg-white/10 text-white/60" title="Show/Hide">
                                    <i id="pin-verify-eye" data-lucide="eye" class="w-4 h-4"></i>
                                </button>
                            </div>
                            <p id="pin-verify-error" class="text-red-400 text-xs mt-2 hidden">Incorrect PIN. Please try again.</p>
                        </div>
                        <div class="flex gap-3">
                            <button type="button" id="pin-cancel" class="flex-1 py-2 bg-white/10 hover:bg-white/20 rounded-lg transition">Cancel</button>
                            <button type="button" id="pin-confirm" class="flex-1 py-2 bg-green-500 hover:bg-green-600 rounded-lg font-medium transition">Confirm</button>
                        </div>
                    </div>
                `;
                container.classList.remove('hidden');
                lucide.createIcons();

                const input = document.getElementById('pin-verify');
                const err = document.getElementById('pin-verify-error');
                const btnConfirm = document.getElementById('pin-confirm');
                const btnCancel = document.getElementById('pin-cancel');
                const btnToggle = document.getElementById('pin-verify-toggle');
                const eye = document.getElementById('pin-verify-eye');

                const close = () => {
                    container.classList.add('hidden');
                    // clear content to avoid stale handlers
                    content.innerHTML = '';
                };

                const verify = () => {
                    const pin = (input?.value || '').trim();
                    // Normalize: keep digits only (helps if user pastes spaces)
                    const normalized = pin.replace(/\s+/g, '');
                    if (!/^\d{4}$/.test(normalized)) {
                        if (err) {
                            err.textContent = 'PIN must be exactly 4 digits.';
                            err.classList.remove('hidden');
                        }
                        return;
                    }
                    const hash = btoa(normalized);
                    const stored = String(emp.pinHash || '').trim();
                    if (hash === stored) {
                        close();
                        return resolve(true);
                    }
                    if (err) {
                        err.textContent = 'Incorrect PIN. Please try again.';
                        err.classList.remove('hidden');
                    }
                    if (input) input.value = '';
                    if (input) input.focus();
                    if (input) input.value = '';
                    if (input) input.focus();
                };

                if (btnCancel) {
                    btnCancel.onclick = () => {
                        close();
                        resolve(false);
                    };
                }
                if (btnConfirm) btnConfirm.onclick = verify;
                if (input) {
                    input.oninput = () => err?.classList.add('hidden');
                    input.onkeydown = (ev) => {
                        if (ev.key === 'Enter') {
                            ev.preventDefault();
                            verify();
                        }
                    };
                    setTimeout(() => input.focus(), 50);
                }
                if (btnToggle && input) {
                    btnToggle.onclick = () => {
                        const hidden = input.type === 'password';
                        input.type = hidden ? 'text' : 'password';
                        if (eye) {
                            eye.setAttribute('data-lucide', hidden ? 'eye-off' : 'eye');
                            lucide.createIcons();
                        }
                    };
                }
            });
        },

                if (err) err.classList.add('hidden');

                if (err) err.classList.remove('hidden');
            }
        },
