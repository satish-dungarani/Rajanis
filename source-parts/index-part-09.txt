        handlePhotoUpload(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (ev) => {
                const img = new Image();
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    const max = 200;
                    let w = img.width, h = img.height;
                    if (w > h) { if (w > max) { h *= max / w; w = max; } }
                    else { if (h > max) { w *= max / h; h = max; } }
                    canvas.width = w;
                    canvas.height = h;
                    canvas.getContext('2d').drawImage(img, 0, 0, w, h);
                    const dataUrl = canvas.toDataURL('image/jpeg', 0.8);
                    document.getElementById('emp-photo-preview').src = dataUrl;
                    this.data.tempPhoto = dataUrl;
                };
                img.src = ev.target.result;
            };
            reader.readAsDataURL(file);
        },

        // ACTIVITY LOG
        renderActivityLog() {
            const table = document.getElementById('activity-table');
            if (!table) return;

            const pageSize = this.data.activityPageSize || 50;
            const search = (this.data.activitySearch || '').toLowerCase();
            const page = Math.max(1, this.data.activityPage || 1);

            const filtered = this.data.history.filter(h => {
                if (!search) return true;
                const name = (h.employeeName || '').toLowerCase();
                const role = (h.employeeRole || '').toLowerCase();
                const action = (h.action || '').toLowerCase();
                return name.includes(search) || role.includes(search) || action.includes(search);
            });

            const total = filtered.length;
            const totalPages = Math.max(1, Math.ceil(total / pageSize));
            this.data.activityPage = Math.min(page, totalPages);

            const start = (this.data.activityPage - 1) * pageSize;
            const end = start + pageSize;
            const pageItems = filtered.slice(start, end);

            const loadingEl = document.getElementById('activity-loading');
            if (loadingEl) loadingEl.classList.remove('hidden');

            table.innerHTML = '';
            if (pageItems.length === 0) {
                table.innerHTML = '<tr><td colspan="6" class="px-4 py-8 text-center text-white/40">No activity records</td></tr>';
            } else {
                pageItems.forEach((h) => {
                    const isIn = h.action.includes('In');
                    const isAuto = h.action.includes('Auto');
                    let badgeClass = isIn ? 'badge-green' : 'badge-red';
                    if (isAuto) badgeClass = 'badge-amber';

                    const hasPhoto = !!h.verificationPhoto;
                    const photoCell = hasPhoto
                        ? `<button type="button" onclick="app.viewVerificationPhoto(${this.data.history.indexOf(h)})" class="group">
                               <img src="${h.verificationPhoto}" class="w-10 h-10 rounded-lg object-cover border border-white/10 bg-black/30 group-hover:opacity-90" loading="lazy" />
                           </button>`
                        : '';

                    table.innerHTML += `<tr>
                        <td class="px-4 py-3 text-white/70">${this.formatDate(h.timestamp)}</td>
                        <td class="px-4 py-3 text-white/70">${this.formatTime(h.timestamp)}</td>
                        <td class="px-4 py-3">${h.employeeName}</td>
                        <td class="px-4 py-3 text-white/50">${h.employeeRole || '-'}</td>
                        <td class="px-4 py-3">${photoCell}</td>
                        <td class="px-4 py-3"><span class="px-2 py-1 rounded text-xs font-medium ${badgeClass}">${h.action}</span></td>
                    </tr>`;
                });
            }

            const rangeEl = document.getElementById('activity-range');
            if (rangeEl) {
                rangeEl.textContent = total === 0
                    ? 'Showing 0-0 of 0'
                    : `Showing ${start + 1}-${Math.min(end, total)} of ${total}`;
            }

            this.renderActivityPagination(totalPages);

            if (loadingEl) {
                setTimeout(() => loadingEl.classList.add('hidden'), 150);
            }
        },

        renderWeeklyEmployeeExcel(emp, weekStart) {
            const { rows, totalHours, maxSessions } = this.buildWeeklyReportData(emp, weekStart);
            const sessionHeaders = [];
            for (let i = 1; i <= Math.max(1, maxSessions); i++) {
                sessionHeaders.push(`<th>In${i}</th><th>Out${i}</th>`);
            }

            const htmlRows = rows.map(r => {
                const cells = [];
                for (let s = 0; s < Math.max(1, maxSessions); s++) {
                    const sess = r.info.sessions[s];
                    if (sess) {
                        cells.push(this.formatTime(sess.in));
                        cells.push(sess.out ? this.formatTime(sess.out) : 'In Progress');
                    } else {
                        cells.push('');
                        cells.push('');
                    }
                }
                return `<tr>
                    <td>${r.dayPlain}${r.isOff ? ' (OFF)' : ''}</td>
                    ${cells.map(c => `<td>${c}</td>`).join('')}
                    <td>${r.hoursValue}</td>
                    <td>${r.noteText}</td>
                </tr>`;
            }).join('');

            return `
                <table>
                    <tr class="employee"><td colspan="${(Math.max(1, maxSessions) * 2) + 4}">Employee: ${emp.name || '-'} (${emp.role || '-'})</td></tr>
                    <tr><td colspan="${(Math.max(1, maxSessions) * 2) + 4}">Week: ${this.formatDate(weekStart)} - ${this.formatDate(new Date(weekStart.getTime() + 6*24*60*60*1000))}</td></tr>
                    <tr></tr>
                    <tr>
                        <th>Day</th>
                        ${sessionHeaders.join('')}
                        <th>Total Hours</th>
                        <th>Notes</th>
                    </tr>
                    ${htmlRows}
                    <tr>
                        <td><strong>Total</strong></td>
                        ${Array(Math.max(1, maxSessions) * 2).fill('<td></td>').join('')}
                        <td><strong>${totalHours.toFixed(2)}</strong></td>
                        <td></td>
                    </tr>
                    <tr>
                        <td class="note">Notes:</td>
                        <td colspan="${(Math.max(1, maxSessions) * 2) + 3}" class="note">Missing clock-out days are excluded from totals. Sessions over 8 hours deduct 30 minutes.</td>
                    </tr>
                </table>
            `;
        },

        renderActivityPagination(totalPages) {
            const container = document.getElementById('activity-pagination');
            if (!container) return;

            container.innerHTML = '';
            if (totalPages <= 1) return;

            const makeBtn = (label, page, disabled = false, active = false) => {
                const btn = document.createElement('button');
                btn.textContent = label;
                btn.className = `px-3 py-1 rounded text-sm ${active ? 'bg-white/20 text-white' : 'bg-white/10 text-white/80 hover:bg-white/20'} ${disabled ? 'opacity-50 cursor-not-allowed' : ''}`;
                if (!disabled) btn.onclick = () => this.setActivityPage(page);
                return btn;
            };

            const prev = makeBtn('Prev', this.data.activityPage - 1, this.data.activityPage <= 1);
            container.appendChild(prev);

            const maxButtons = 5;
            let start = Math.max(1, this.data.activityPage - 2);
            let end = Math.min(totalPages, start + maxButtons - 1);
            if (end - start < maxButtons - 1) {
                start = Math.max(1, end - maxButtons + 1);
            }

            for (let i = start; i <= end; i++) {
                container.appendChild(makeBtn(String(i), i, false, i === this.data.activityPage));
            }

            const next = makeBtn('Next', this.data.activityPage + 1, this.data.activityPage >= totalPages);
            container.appendChild(next);
        },

        applyCustomDateRange() {
            const startPicker = document.getElementById('custom-start-date-picker');
            const endPicker = document.getElementById('custom-end-date-picker');
            const rangeInput = document.getElementById('custom-date-range');
            const hs = document.getElementById('custom-start-date');
            const he = document.getElementById('custom-end-date');
            if (!startPicker || !endPicker || !rangeInput || !hs || !he) return;

            const startVal = startPicker.value;
            const endVal = endPicker.value;
            // Set hidden fields
            hs.value = startVal || '';
            he.value = endVal || '';

            // Validate immediately to enforce Sunday/Saturday
            if (!this.validateCustomDates()) return;

            // Display range in text box
            if (startVal && endVal) {
                const sDate = new Date(startVal);
                const eDate = new Date(endVal);
                const fmt = d => `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}/${d.getFullYear()}`;
                rangeInput.value = `${fmt(sDate)} - ${fmt(eDate)}`;
            } else {
                rangeInput.value = '';
            }
            // Hide panel
            const panel = document.getElementById('custom-date-range-panel');
            if (panel) panel.classList.add('hidden');
        },

        validateCustomDates() {
            const start = document.getElementById('custom-start-date')?.value;
            const end = document.getElementById('custom-end-date')?.value;
            const err = document.getElementById('custom-date-error');
            if (!start || !end) {
                err?.classList.add('hidden');
                return true;
            }
            const s = new Date(start);
            const e = new Date(end);
            const valid = s.getDay() === 0 && e.getDay() === 6 && s <= e;
            if (!valid) err?.classList.remove('hidden'); else err?.classList.add('hidden');
            return valid;
        },
