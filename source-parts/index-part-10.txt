        exportCustomReportExcel() {
            const startVal = document.getElementById('custom-start-date')?.value;
            const endVal = document.getElementById('custom-end-date')?.value;
            const empIdRaw = document.getElementById('custom-emp-id')?.value;
            if (!startVal || !endVal || !empIdRaw) {
                this.uiToast('Generate the report first', 'error');
                return;
            }
            const weekStart = new Date(startVal);
            const weekEnd = new Date(endVal);

            let employees = [];
            if (empIdRaw === 'all') {
                employees = this.data.employees.filter(e => e.active !== false);
            } else {
                const emp = this.data.employees.find(e => e.id === parseInt(empIdRaw));
                if (!emp) { this.uiToast('Employee not found', 'error'); return; }
                employees = [emp];
            }

            // Determine max sessions across selected employees for table structure
            let maxSessions = 0;
            const byEmp = new Map();
            employees.forEach(emp => {
                const byDay = this.getWeeklySessionsForEmployee(emp.id, weekStart);
                byEmp.set(emp.id, byDay);
                Object.values(byDay).forEach(info => {
                    const c = (info.sessions || []).length;
                    if (c > maxSessions) maxSessions = c;
                });
            });
            maxSessions = Math.max(1, maxSessions);
            const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];

            let html = `<!DOCTYPE html><html><head><meta charset="UTF-8">
                <style>
                    body { font-family: Arial, sans-serif; }
                    table { border-collapse: collapse; width: 100%; font-size: 12px; }
                    th, td { border: 1px solid #333; padding: 6px; text-align: left; }
                    th { background: #f1f5f9; font-weight: bold; }
                    .title { background: #e2e8f0; font-weight: bold; font-size: 14px; }
                    .employee { background: #eef2ff; font-weight: bold; }
                    .note { color: #6b7280; font-size: 11px; }
                </style>
            </head><body>`;

            html += `<table>`;
            html += `<tr class="title"><th colspan="${(maxSessions * 2) + 4}">Custom Report</th></tr>`;
            html += `<tr><td><strong>Range</strong></td><td colspan="${(maxSessions * 2) + 3}">${this.formatDate(weekStart)} - ${this.formatDate(weekEnd)}</td></tr>`;
            html += `<tr></tr>`;

            employees.forEach(emp => {
                const byDay = byEmp.get(emp.id) || {};
                let empTotal = 0;

                html += `<tr class="employee"><td colspan="${(maxSessions * 2) + 4}">Employee: ${emp.name || '-'} (${emp.role || '-'})</td></tr>`;

                const sessionHeaders = [];
                for (let i = 1; i <= maxSessions; i++) {
                    sessionHeaders.push(`<th>In${i}</th><th>Out${i}</th>`);
                }
                html += `<tr><th>Day</th>${sessionHeaders.join('')}<th>Total Hours</th><th>Notes</th></tr>`;

                for (let i = 0; i < 7; i++) {
                    const dayDate = new Date(weekStart);
                    dayDate.setDate(dayDate.getDate() + i);
                    const key = dayDate.toDateString();
                    const info = byDay[key] || { sessions: [], totalHours: 0, missingClockOut: false, deductionNote: '' };

                    const cells = [];
                    for (let s = 0; s < maxSessions; s++) {
                        const sess = info.sessions[s];
                        if (sess) {
                            cells.push(this.formatTime(sess.in));
                            cells.push(sess.out ? this.formatTime(sess.out) : 'In Progress');
                        } else {
                            cells.push('');
                            cells.push('');
                        }
                    }

                    const hours = info.missingClockOut ? '' : info.totalHours.toFixed(2);
                    const note = info.missingClockOut ? 'Missing clock-out' : (info.deductionNote || '');
                    if (!info.missingClockOut) empTotal += info.totalHours;

                    html += `<tr>`;
                    html += `<td>${dayNames[i]} ${this.formatDate(dayDate)}</td>`;
                    html += cells.map(c => `<td>${c}</td>`).join('');
                    html += `<td>${hours}</td>`;
                    html += `<td>${note}</td>`;
                    html += `</tr>`;
                }

                html += `<tr><td><strong>Total</strong></td>${Array(maxSessions * 2).fill('<td></td>').join('')}<td><strong>${empTotal.toFixed(2)}</strong></td><td></td></tr>`;
                html += `<tr><td class="note">Notes:</td><td colspan="${(maxSessions * 2) + 3}" class="note">Missing clock-out days are excluded from totals. Sessions over 8 hours deduct 30 minutes.</td></tr>`;
                html += `<tr></tr>`;
            });

            html += `</table></body></html>`;

            const blob = new Blob([html], { type: 'application/vnd.ms-excel' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            const empName = employees.length === 1 ? employees[0].name.toLowerCase().replace(/\s+/g, '_') : 'all';
            a.href = url;
            a.download = `custom_report_${empName}.xls`;
            a.click();
            URL.revokeObjectURL(url);
        },

        // Build sessions per day for a given employee + week.
        // Sessions are pairs of: Clocked In -> next Clocked Out / Auto Clock Out.
        // Supports multiple sessions in the same day.
        getWeeklySessionsForEmployee(empId, weekStart) {
            const emp = this.data.employees.find(e => e.id === empId);
            const byDay = {};
            if (!emp) return byDay;

            for (let i = 0; i < 7; i++) {
                const day = new Date(weekStart);
                day.setDate(day.getDate() + i);
                const dayStart = new Date(day);
                dayStart.setHours(0, 0, 0, 0);
                const dayEnd = new Date(dayStart);
                dayEnd.setDate(dayEnd.getDate() + 1);

                const dayRecords = this.data.history
                    .filter(h => h.employeeId === empId)
                    .filter(h => {
                        const t = new Date(h.timestamp);
                        return t >= dayStart && t < dayEnd;
                    })
                    .slice()
                    .sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));

                    let openIn = null;
                    let missingClockOut = false;
                    let deductionMinutes = 0;
                    const sessions = [];

                    for (const r of dayRecords) {
                        if (r.action === 'Clocked In') {
                            if (!openIn) openIn = new Date(r.timestamp);
                        } else if (r.action === 'Clocked Out' || r.action === 'Auto Clock Out (System)') {
                            if (openIn) {
                                const out = new Date(r.timestamp);
                                if (out > openIn) sessions.push({ in: openIn, out });
                                openIn = null;
                            }
                        }
                    }

                    // If still clocked in today, show an in-progress session
                    if (openIn && emp.status === 'in' && day.toDateString() === new Date().toDateString()) {
                        sessions.push({ in: openIn, out: null, inProgress: true });
                    }

                    // Missing clock-out rule: if any open session exists, mark missing (no hours counted)
                    if (sessions.some(s => !s.out)) {
                        missingClockOut = true;
                    }

                    let totalHours = 0;

                    if (!missingClockOut) {
                        sessions.forEach(session => {
                            if (!session.out) return;
                            let sessionHours = Math.max(0, (session.out - session.in) / (1000 * 60 * 60));
                            if (sessionHours > 8) {
                                sessionHours = Math.max(0, sessionHours - 0.5);
                                deductionMinutes += 30;
                            }
                            totalHours += sessionHours;
                        });
                    }

                    const deductionNote = deductionMinutes > 0 ? `${deductionMinutes} min break deducted` : '';
                    byDay[day.toDateString()] = { sessions, totalHours, missingClockOut, deductionNote, deductionMinutes };

                // Store only for weekly/custom reports (NOT written to DB)
                const mapKey = `${empId}|${new Date(weekStart).toDateString()}`;
                this.data.photoMap[mapKey] = this.data.photoMap[mapKey] || {};
                this.data.photoMap[mapKey][day.toDateString()] = (dayRecords || []).map(r => r.verificationPhoto).filter(Boolean);
            }

            return byDay;
        },

        buildWeeklyReportData(emp, weekStart) {
            const rows = [];
            const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
            const byDay = this.getWeeklySessionsForEmployee(emp.id, weekStart);
            let totalHours = 0;
            let maxSessions = 0;

            for (let i = 0; i < 7; i++) {
                const day = new Date(weekStart);
                day.setDate(day.getDate() + i);
                const key = day.toDateString();
                const info = byDay[key] || { sessions: [], totalHours: 0, missingClockOut: false, deductionNote: '' };
                const isOff = (emp.offDays || []).includes(i);
                const dayLabel = `${dayNames[i]}, ${this.formatDate(day)}`;
                const sessionsCount = info.sessions.length;
                maxSessions = Math.max(maxSessions, sessionsCount);

                const sessionsHtml = sessionsCount
                    ? info.sessions
                        .map(s => `${this.formatTime(s.in)}-${s.out ? this.formatTime(s.out) : '<span class="badge-amber px-2 py-0.5 rounded text-xs">In Progress</span>'}`)
                        .join(' <span class="text-white/30">|</span> ')
                    : '<span class="text-white/40">-</span>';

                let hoursDisplay = '<span class="text-white/40">-</span>';
                if (!info.missingClockOut) {
                    const hoursVal = info.totalHours.toFixed(2);
                    hoursDisplay = `<span class="${info.totalHours >= 8 ? 'text-green-400' : 'text-white/70'} font-mono">${hoursVal}</span>`;
                    totalHours += info.totalHours;
                }

                let noteHtml = '<span class="text-white/30 text-xs">-</span>';
                let noteText = '';
                if (info.missingClockOut) {
                    noteHtml = '<span class="text-red-400 text-xs">Missing clock-out</span>';
                    noteText = 'Missing clock-out';
                } else if (info.deductionNote) {
                    noteHtml = `<span class="text-amber-400 text-xs">${info.deductionNote}</span>`;
                    noteText = info.deductionNote;
                }

                rows.push({
                    dayLabel,
                    dayPlain: `${dayNames[i]} ${this.formatDate(day)}`,
                    sessionsHtml,
                    notesHtml: noteHtml,
                    noteText,
                    hoursDisplay,
                    hoursValue: info.missingClockOut ? '' : info.totalHours.toFixed(2),
                    rowClass: isOff ? 'bg-amber-500/10' : '',
                    offBadge: isOff ? '<span class="badge-amber px-2 py-0.5 rounded text-xs ml-2">OFF</span>' : '',
                    info,
                    dayDate: day,
                    isOff
                });
            }

            return { rows, totalHours, maxSessions };
        },

            const empId = parseInt(document.getElementById('weekly-emp-id')?.value);
            const dateVal = document.getElementById('weekly-date')?.value;
            if (!empId || !dateVal) return;

                </tr>`;
            });

        },
