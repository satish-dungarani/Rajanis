        async dbReadEmployeesAndHistory() {
            const src = this.getDataSource();
            if (src === 'supabase') {
                this.showLoader('Loading data...');
                try {
                    const { data: employees, error: empErr } = await this.data.supabase.from('employees').select('*');
                    if (empErr) throw empErr;
                    const { data: history, error: histErr } = await this.data.supabase.from('history').select('*').order('timestamp', { ascending: false });
                    if (histErr) throw histErr;
                    return { employees: (employees || []).map(this.mapEmployeeFromDB), history: (history || []).map(this.mapHistoryFromDB) };
                } catch (e) {
                    this.logError('dbReadEmployeesAndHistory', e);
                    throw e;
                } finally {
                    this.hideLoader();
                }
            }
            return { employees: [], history: [] };
        },

        async syncEmployeesToSupabase() {

        initSupabase() {
            const cfg = this.getSupabaseConfig();
            const url = (cfg?.url || '').trim();
            const key = (cfg?.key || '').trim();
            if (!url || !key || !window.supabase?.createClient) {
                this.data.supabase = null;
                return;
            }
            try {
                this.data.supabase = window.supabase.createClient(url, key, { auth: { persistSession: false } });
                this.updateSupabaseStatus(true);
            } catch (e) {
                this.logError('initSupabase', e);
                this.data.supabase = null;
            }
        },

        async testSupabaseConnection() {
            this.initSupabase();
            if (!this.data.supabase) {
                this.uiAlert('Please enter Supabase URL and Key first.');
                return;
            }
            try {
                const { error } = await this.data.supabase.from('employees').select('id').limit(1);
                if (error) throw error;
                this.uiToast('Supabase connection successful', 'success');
                this.updateSupabaseStatus(true);
            } catch (e) {
                console.error(e);
                this.updateSupabaseStatus(false);
                this.uiAlert('Supabase connection failed. Check URL, Key, and schema.');
            }
        },

        async loadSupabaseSettings() {
            if (!this.data.supabase) return;
            try {
                const { data, error } = await this.data.supabase.from('app_settings').select('*');
                if (error) throw error;
                const map = {};
                data.forEach(row => { map[row.key] = row.value; });
                if (map.branding) { this.data.branding = map.branding; this.applyBranding(); }
                if (map.theme) { this.data.currentTheme = map.theme; this.applyTheme(); }
                if (map.locationSettings) {
                    this.data.locationSettings = map.locationSettings;
                    this.data.gpsRestrictionEnabled = !!map.locationSettings.gpsEnabled;
                    this.data.ipRestrictionEnabled = !!map.locationSettings.ipEnabled;
                    this.renderLocationSettings();
                }
                if (map.pinVerificationEnabled !== undefined) this.data.pinVerificationEnabled = !!map.pinVerificationEnabled;
                if (map.cameraVerificationEnabled !== undefined) this.data.cameraVerificationEnabled = !!map.cameraVerificationEnabled;
                if (map.adminPasswordHash) this.data.adminPasswordHash = String(map.adminPasswordHash);
                this.updateVerificationSettingsUI();
                this.updateSupabaseStatus(true);
                this.data.settingsLoaded = true;
                if (this.data.adminView === 'settings') {
                    this.updateVerificationSettingsUI();
                }
            } catch (e) {
                console.error('Supabase settings load error', e);
            }
        },

        async saveSupabaseSetting(key, value) {
            if (!this.data.supabase) return { error: 'no_supabase' };
            const { error } = await this.data.supabase
                .from('app_settings')
                .upsert({ key, value }, { onConflict: 'key' });
            if (error) {
                console.error('Supabase setting save error:', error);
                this.uiToast('Failed to save setting', 'error');
            }
            return { error };
        },

            const sql = this.getSupabaseSQL();

                this.uiToast('SQL copied to clipboard', 'success');
            });
        },

        getSupabaseSQL() {
            return `-- Employees\ncreate table if not exists employees (\n  id bigint primary key,\n  name text not null,\n  role text not null,\n  gender text,\n  mobile text,\n  email text,\n  photo text,\n  off_days int[],\n  status text,\n  last_action timestamptz,\n  active boolean default true,\n  display_order int,\n  pin_hash text\n);\n\n-- History\ncreate table if not exists history (\n  id bigint generated always as identity primary key,\n  employee_id bigint references employees(id),\n  employee_name text,\n  employee_role text,\n  action text,\n  timestamp timestamptz,\n  verification_photo text\n);\n\n-- Settings\ncreate table if not exists app_settings (\n  key text primary key,\n  value jsonb\n);\n\n-- Backups\ncreate table if not exists backups (\n  id bigint generated always as identity primary key,\n  created_at timestamptz default now(),\n  data jsonb\n);`;
        },
